import java.math.BigInteger;
import java.util.Scanner;

public class Main {
    //n개의 원반을 x기둥에서 y기둥으로 옮긴다
	public static void Hanoi1(int n, int x, int y) { 
		if(n == 0)
			return;
		Hanoi1(n-1, x, 6-x-y); //n-1개의 원반을 중간 기둥으로 옮긴다
		System.out.println(x + " " + y); //가장 밑바닥 원반을 시작 기둥에서 목표 기둥으로 옮겼음 출력
		Hanoi1(n-1, 6-x-y, y); //n-1개의 원반을 중간기둥에서 목표지점 기둥으로 옮긴다
	}
	public static void main(String[] args) {
		Scanner sc = new Scanner(System.in);
		int n = sc.nextInt(); //원판 개수 입력받음
		BigInteger count = BigInteger.valueOf(2).pow(n); 
		BigInteger result = count.subtract(BigInteger.ONE); 
		System.out.println(result);//원반 이동횟수 출력 (2^n-1)
		
		if(n<=20) {
			Hanoi1(n,1,3);
		}		
	}
}
